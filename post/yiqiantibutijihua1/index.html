Codeforces Round #798 (Div. 2) E. ANDfinity 题解

E. ANDfinity

题意

给定一个长度为nnn的数组a(ai≥0)a (a_i\geq0)a(ai​≥0)，要求构造一个nnn个顶点的图，对于每对节点(i,j)(i,j)(i,j)，当且仅当ai&aj>0a_i \& a_j>0ai​&aj​>0时，两个节点连接一条边。

现有两种操作：

  1. 选择一个数使其增 111
  2. 选择一个数使其减 111

求能使得nnn个顶点连通的最小操作数

输入：t,n,a1…n(1≤t≤1000,1≤n≤2000,0≤ai<230)t,n,a_{1\dots n}(1\leq t\leq 1000,1\leq n\leq 2000,0\leq a_i<2^{30})t,n,a1…n​(1≤t≤1000,1≤n≤2000,0≤ai​<230)

输出：最小使得图连通的操作次数

思路

对于任意一个数组，只需找到一个lowbit(ai)lowbit(a_i)lowbit(ai​)最大的元素aia_iai​，使其−1-1−1，则所有lowbit(aj)(j!=i)lowbit(a_j)(j!=i)lowbit(aj​)(j!=i)小于lowbit(ai)lowbit(a_i)lowbit(ai​)的元素都会与aia_iai​连通；若有多个lowbitlowbitlowbit相同的元素，使一个−1-1−1，剩下的其中一个+1+1+1即可连通所有元素。所以最大操作次数是两次。

特判：对于所有数lowbitlowbitlowbit相同，仅有一个数lowbitlowbitlowbit小一位的情况，需要特判，因为这种情况可以直接让lowbitlowbitlowbit小一位的数+1+1+1，就能一步使整个图连通。

复杂度：O(n2log(ai))O(n^2log(a_i))O(n2log(ai​))

对每个元素操作一次时检查连通性，遍历每个元素是O(n)O(n)O(n)，检查连通性是O(nlog(ai))O(nlog(a_i))O(nlog(ai​))

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
int lowbit(int x){return (x)&(-x);}
const int MAX_N = 2010;

int T;
int n,ans;
int a[MAX_N];
//并查集 
int f[40];

int find(int x){
	if(f[x]==x)
		return x;
	return f[x]=find(f[x]);
}
//检查整个图是否连通 
int check(){
	//并查集初始化 
	for(int i=0;i<=30;i++)
		f[i]=i;
	//把所有有1的位记录到flag，最后用于遍历 
	int flag=0;
	for(int i=1;i<=n;i++)
		flag|=a[i];
	//遍历每个元素，把每个元素的所有为1的位连通 
	for(int i=1;i<=n;i++){
		int last=-1;
		//遍历每一位 
		for(int j=0;j<=30;j++){
			//如果当前位是1 
			if((a[i]>>j)&1){
				//如果不是第一个位为1的，和第一个加入同一集合 
				if(last!=-1)
					f[find(last)]=find(j);
				last=j;
			}
		}
	}
	//检查整个图是否连通 
	set<int> se;
	//遍历每一位
	for(int i=0;i<=30;i++)
		//如果这一位有1 
		if((flag>>i)&1)
			//插入该位所在集合祖先 
			se.insert(find(i));
	//如果集合只有一个元素，说明整个图是连通的 
	return se.size()==1;
}

void print(){
	cout<<ans<<'\n';
	for(int i=1;i<=n;i++)
		cout<<a[i]<<' ';
	cout<<'\n';
}

void solve(){
	ans=0;
	cin>>n;
	int mxb=0;
	for(int i=1;i<=n;++i){
		cin>>a[i];
		//如果a[i]=0，是一定要操作一次+1使其变为1的 
		if(a[i]==0){
			ans++;
			a[i]=1;
		}
		//记录lowbit最大位 
		mxb=max(mxb,lowbit(a[i]));
	}
	//检查整个图是否连通 
	if(check()){
		//连通就直接输出 
		print();
		return ;
	}
	//初始不连通，则特判是否存在一位操作一次-1就能连通 
	ans++;
	for(int i=1;i<=n;i++){
		if(a[i]==1)
			continue;
		//先-1 
		a[i]--;
		//检查整个图是否连通 
		if(check()){
			print();
			return ;
		}
		//不连通再+1 
		a[i]++;
	}
	//再特判是否存在一位操作一次+1就能连通 
	for(int i=1;i<=n;i++){
		//先+1 
		a[i]++;
		//检查整个图是否连通 
		if(check()){
			print();
			return ;
		}
		//不连通再-1 
		a[i]--;
	}
	//如果一次操作不行，那就一定是两次
	ans++;
	//先让lowbit最大的位-1 
	for(int i=1;i<=n;i++)
		if(lowbit(a[i])==mxb){
			a[i]--;
			break;
		}
	//再让lowbit并列最大的位+1 
	for(int i=1;i<=n;i++)
		if(lowbit(a[i])==mxb){
			a[i]++;
			break;
		}
	//输出答案 
	print();
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		solve();
	} 
	return 0;
}
