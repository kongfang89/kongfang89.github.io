D. Potion Brewing Class

题意

给定nnn种药和n−1n-1n−1种配比关系，求最少需要多少总剂量可以使得所有药的需求均为整数，且满足所给配比关系。

输入：t,n,u,v,x,y(1≤t≤104,1≤n≤2×105,1≤u,v≤n,1≤x,y≤n)t,n,u,v,x,y(1\leq t \leq 10^4,1\leq n\leq 2\times 10^5,1\leq u,v\leq n,1\leq x,y\leq n)t,n,u,v,x,y(1≤t≤104,1≤n≤2×105,1≤u,v≤n,1≤x,y≤n)

输出：最少能满足所有配比关系的总剂量（对998244353998244353998244353取模）

思路

建树，任选一点作为根出发深搜求得满足该点的最小剂量，再根据配比关系深搜求得所有药的最小剂量。

如何求根节点的最小剂量？

对于两个分数关系，求其分母的lcmlcmlcm在约去分子的公因子即可。

一个结论：一堆数的lcmlcmlcm等于它们分解质因数后，每个质因数的幂次最大的乘积。

复杂度：O(nlogn)O(nlogn)O(nlogn)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
#define YES cout<<"YES"<<endl
#define NO cout<<"NO"<<endl
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const ll mod=998244353;
const ll MAX_N = 2e5+10;

int T=1;
//质因数 
vector<pair<int,int> > fac[MAX_N];
//节点 
struct node{
	int v,x,y;
};
//树 
vector<node> G[MAX_N];
//n个点，每个点的值，质因数出现的最大次数，质因数的次数 
int n,a[MAX_N],mx[MAX_N],cot[MAX_N]; 
int ans=0;

ll ksm(ll x,ll n){
	ll res=1;
	while(n){
		if(n&1)
			res=(res*x)%mod;
		x=x*x%mod;
		n>>=1;
	}
	return res;
}

void dfs(int u,int fa){
	//遍历所有子节点 
	for(auto i:G[u]){
		int v=i.v,x=i.x,y=i.y;
		if(v==fa)
			continue;
		//计算分母的质因数给的最大值 
		for(auto p:fac[x]){
			cot[p.first]+=p.second;
			mx[p.first]=max(mx[p.first],cot[p.first]);
		}
		//用分子的质因子消去一些 
		for(auto p:fac[y])
			cot[p.first]-=p.second;
		dfs(v,u);
		//回溯 
		for(auto p:fac[y])
			cot[p.first]+=p.second;
		for(auto p:fac[x])
			cot[p.first]-=p.second;
	}
}

void dfs2(int u,int fa){
	//统计答案 
	ans=(ans+a[u])%mod;
	//更新子节点的值 
	for(auto i:G[u]){
		int v=i.v,x=i.x,y=i.y;
		if(v==fa)
			continue;
		a[v]=((a[u]*ksm(x,mod-2))%mod*y)%mod;
		dfs2(v,u);
	}
}

signed main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	
	//预处理出2~200000每个数的质因数 
	for(int i=2;i<=200000;i++){
		int t=i;
		for(int j=2;j<=sqrt(i);j++){
			int cnt=0;
			while(t%j==0){
				t/=j;
				cnt++; 
			}
			if(cnt)
				fac[i].push_back({j,cnt});
		}
		if(t!=1)
			fac[i].push_back({t,1});
	}
	cin>>T;
	while(T--){
		cin>>n;
		for(int i=0;i<=n;i++){
			G[i].clear();
			mx[i]=0;
			cot[i]=0;
		}
		//建树 
		for(int i=1;i<n;i++){
			int u,v,x,y;
			cin>>u>>v>>x>>y;
			cout<<x<<' '<<y<<'\n';
			int g=__gcd(x,y);
			//求lcm 
			x=x/g,y=y/g;
			cout<<x<<' '<<y<<'\n';
			G[u].push_back({v,x,y});
			G[v].push_back({u,y,x});
		}
		dfs(1,0);
		int val=1;
		for(int i=0;i<=n;i++)
			if(mx[i])
				val=(val*ksm(i,mx[i]))%mod;
		a[1]=val;
		ans=0;
		dfs2(1,0);
		cout<<ans<<'\n';
	}
	return 0;
}
