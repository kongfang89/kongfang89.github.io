E. Arithmetic Operations

题意

给定nnn个数，先可修改aia_iai​为任意整数，问最少修改多少次可以使得这nnn个数是等差数列。

输入：n,a1…n(1≤n≤105,1≤ai≤105)n,a_{1\dots n}(1\leq n\leq 10^5,1\leq a_i\leq 10^5)n,a1…n​(1≤n≤105,1≤ai​≤105)

输出：最少修改次数

思路

根号分治题。

设公差为ddd，则有ai=a0+i∗da_i=a_0+i*dai​=a0​+i∗d

我们将情况分为两种：

**Case 1 : d≤nd\leq \sqrt{n}d≤n​ **

此时我们枚举所有d[−n,n]d[-\sqrt{n},\sqrt{n}]d[−n​,n​]，由可ai=a0+i∗da_i=a_0+i*dai​=a0​+i∗d知a0=ai−i∗da_0=a_i-i*da0​=ai​−i∗d，则每次枚举我们都可以求得每个数所对应的等差数列首项，只需要在枚举的同时记录最常出现的等差数列首项即可

复杂度：O(nn)O(n\sqrt{n})O(nn​)

**Case 2 : d>nd>\sqrt{n}d>n​ **

由于公差d>nd>\sqrt{n}d>n​，则原数列满足该公差的项数不会太多，则我们枚举数列中每项数作为首项的合法情况，统计最长的等差数列即可

复杂度：O(nn)O(n\sqrt{n})O(nn​)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
#define YES cout<<"YES"<<endl
#define NO cout<<"NO"<<endl
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int mod= 1e9+7;
const int s=405;
const int MAX_N = 1e5+10;
const int mi=MAX_N*s;

int n;
int a[MAX_N];
int vis[MAX_N*(s*2+10)]; 

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    int ma=1;
    for(int d=-s;d<=s;d++){
        for(int k=1;k<=n;k++){
            int kn=a[k]-k*d+mi;
            vis[kn]++;
            ma=max(ma,vis[kn]);
        }
        for(int i=1;i<=n;i++){
            int kn=a[i]-i*d+mi;
            vis[kn]=0;
        }
    } 
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=i+s&&j<=n;j++){
            int cha=a[j]-a[i];
            if(cha%(j-i))
                continue;
            int now=cha/(j-i)+mi;
            vis[now]++;
            ma=max(ma,vis[now]+1);
        }
        for(int j=i+1;j<=i+s&&j<=n;j++){
            int cha=a[j]-a[i];
            if(cha%(j-i))
                continue;
            int now=cha/(j-i)+mi;
            vis[now]=0;
        }
    }
    cout<<n-ma<<'\n';
	return 0;
}
