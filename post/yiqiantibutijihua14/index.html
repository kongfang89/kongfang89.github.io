E. Star MST

题意

现有一nnn个节点的无向完全图，边权w∈[1,k]w\in[1,k]w∈[1,k]，已知该图最小生成树的权值和与所有与节点111相连的边权和相等，问有多少构图方式的最小生成树满足要求

输入：n,k(2≤n≤250,1≤k≤250)n,k(2\leq n\leq 250,1\leq k\leq 250)n,k(2≤n≤250,1≤k≤250)

输出：方案数（对998244353998244353998244353取模）

思路

二维dp

dp[i][j]dp[i][j]dp[i][j]表示选了jjj个节点(不包括111)，最小生成树最大边权为iii时的方案数，则答案为dp[k][n−1]dp[k][n-1]dp[k][n−1]

状态转移：对于状态dp[i−1][j]dp[i-1][j]dp[i−1][j]，此时最小生成树最大边权为i−1i-1i−1，仍有n−j−1n-j-1n−j−1个节点未选，则我们可以枚举选aaa个节点与节点111相连的边权为iii，方案数为Cn−j−1aC_{n-j-1}^aCn−j−1a​。由最小生成树的构造方式得，这aaa个节点与其它不是节点111的节点的边权必须≥i\geq i≥i(否则选其它边更优)，这样的边有j×a+a(a−1)2j\times a+\frac{a(a-1)}{2}j×a+2a(a−1)​种(与jjj个节点分别相连+内部相连)，可选的边权有k−i+1k-i+1k−i+1种，则边权方案数为(k−i+1)j×a+a(a−1)2(k-i+1)^{j\times a+\frac{a(a-1)}{2}}(k−i+1)j×a+2a(a−1)​

复杂度：O(n2klogn)O(n^2klogn)O(n2klogn)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
#define YES cout<<"YES"<<endl
#define NO cout<<"NO"<<endl
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int mod= 998244353;
const int MAX_N = 300;

int n,k;
ll fac[MAX_N],inv[MAX_N];
ll dp[MAX_N][MAX_N]; 
//快速幂 
ll ksm(ll a,ll n){
	ll res=1;
	while(n){
		if(n&1)
			res=res*a%mod;
		a=a*a%mod;
		n>>=1;
	}
	return res;
}
//预处理阶乘和逆元 
void init(){
	fac[0]=inv[0]=1;
	for(int i=1;i<MAX_N;i++)
		fac[i]=(fac[i-1]*i)%mod;
	inv[MAX_N-1]=ksm(fac[MAX_N-1],mod-2);
	for(int i=MAX_N-2;i>=1;i--)
		inv[i]=inv[i+1]*(i+1)%mod; 
}
//计算组合数 
ll C(int a,int b){
	return fac[a]*inv[b]%mod*inv[a-b]%mod;
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	init();
	cin>>n>>k;
	dp[0][0]=1;
	//枚举边权 
	for(int i=1;i<=k;i++)
		//枚举选点数 
		for(int j=0;j<n;j++)
			//枚举再选a个节点的状态转移 
			for(int a=0;a<n-j;a++)
				dp[i][j+a]=(dp[i][j+a]+dp[i-1][j]*C(n-j-1,a)%mod*ksm(k-i+1,j*a+a*(a-1)/2)%mod)%mod;
	cout<<dp[k][n-1]<<'\n';
	return 0;
}
