F. Juju and Binary String

题意

给定一个长度为nnn的010101串sss，定义可爱度为010101串中的111的个数比上串长度，问是否有若干个不相交的子串满足：子串长度和为mmm；子串拼起来后的可爱度与所给串sss一样

输入：t,n,m,s(1≤t≤104,1≤m≤n≤2×105)t,n,m,s(1\leq t\leq 10^4,1\leq m\leq n\leq 2\times 10^5)t,n,m,s(1≤t≤104,1≤m≤n≤2×105)

输出：符合要求的所有子串的左右端点

思路

巧妙的思维题。

字符串有AAA个000和BBB个111，则记000的价值为−B-B−B，111的价值为AAA，则所给串sss的价值为000，问题就转换为求是否有若干个子串拼起来后长度为mmm且总价值为000。

设串sss的可爱度为an\frac{a}{n}na​，则所求子串拼起来的串ttt中111的个数为amn\frac{am}{n}nam​，只要这个数为整数就有解。

结论：将字符串首尾相连，一定有一个长度为mmm的子串价值为000

证明：如果每个区间的价值都为正或都为负，则整个串的价值一定都为正或都为负，则一定有两个串的价值异号，由于相邻的串的111的数量变化不会超过111，故由正到负或由负到正一定会经过价值为000的串（类似零点定理）

故只要把串复制一遍，计算价值前缀和，扫一遍看看哪个长度为mmm的区间价值为000即可，如果区间经过了nnn，拆成两段输出。

复杂度：O(n)O(n)O(n)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
#define YES cout<<"YES"<<endl
#define NO cout<<"NO"<<endl
#define int long long
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int mod= 1e9+7;
const int MAX_N = 2e5+10;

int T=1;
int n,m;
char s[MAX_N];
int a[MAX_N*2];

signed main(){
	freopen("1.in", "r", stdin);
	//ios_base::sync_with_stdio(false);
	//cout.tie(0);
	//cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		cin>>n>>m;
		cin>>s+1;
		int cnt=0;
		for(int i=1;i<=n;i++)
			if(s[i]=='1')
				cnt++;
		if((m*cnt)%n){
			cout<<-1<<'\n';
			continue;
		}
        //复制一遍
		for(int i=1;i<=n;i++){
			a[i]=s[i]-'0';
			a[i+n]=s[i]-'0';
		}
		int nn=n*2;
        //计算每个点的价值，1的价值为A，0的价值为-B
		for(int i=1;i<=nn;i++){
			if(a[i])
				a[i]=n-cnt;
			else
				a[i]-=cnt;
		}
        //计算价值前缀和
		for(int i=1;i<=nn;i++)
			a[i]+=a[i-1];
		int l=0,r=0;
        //从头扫一遍看看哪个长度为m的区间价值为0
		for(int i=m;i<=nn;i++){
			if(a[i]-a[i-m]==0){
				l=i-m+1;
				r=i;
				break;
			}
		}
        //判断有没有经过n，经过了就拆成两段
		if(r<=n){
			cout<<1<<'\n';
			cout<<l<<' '<<r<<'\n';
		}
		else{
			cout<<2<<'\n';
			cout<<1<<' '<<r-n<<'\n';
			cout<<l<<' '<<n<<'\n';
		}
	}
	return 0;
}
