E. Hemose on the Tree

题意

给定一棵nnn个顶点的树，其中n=2pn=2^pn=2p，现要求将[1,2n−1][1,2n-1][1,2n−1]赋值给nnn个节点和n−1n-1n−1条边，在赋值后，需选定一个节点作为根，使得从根出发到每个节点或每条边的最大成本值最小化。

根到每个节点或每条边的成本定义为：这条路径上所有点和边的值的异或和。

输入：t,p,u,v(1≤t≤50000,1≤p≤17,1≤u,v≤n)t,p,u,v(1\leq t\leq 50000,1\leq p\leq 17,1\leq u,v\leq n)t,p,u,v(1≤t≤50000,1≤p≤17,1≤u,v≤n)

输出：所选根，每个节点的值，每条边的值

思路

由于ansansans是[1,2n−1][1,2n-1][1,2n−1]某些数的异或和，故ans∈[1,2n−1]ans\in[1,2n-1]ans∈[1,2n−1]。如果ans<nans<nans<n，则根节点值必须<n<n<n，且所有点和边的值必须<n<n<n，依据题意所给区间，无解。故ans≥nans\geq nans≥n，此时分析是否存在一种赋值方案使答案最小，即ans=nans=nans=n。对于1≤i<n1\leq i<n1≤i<n，均有(i+n)⊗n=i(i+n)\otimes n=i(i+n)⊗n=i，故将所有i,i+ni,i+ni,i+n配对。

赋值方案：从根节点出发，如果当前异或和为nnn，则所连边赋值i+ni+ni+n，所连点赋值nnn，则到所连点的异或和为000；如果当前异或和为000，则所连边赋值nnn，所连点赋值i+ni+ni+n，则到所连点的异或和为nnn。

复杂度：O(n)O(n)O(n)

只需从任意节点出发dfs遍历整棵树同时赋值即可。

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = (1<<17)+10;

int T;
int n;
int a[MAX_N];
vector<pair<int,int> > edge;
map<pair<int,int>,int> mp;
vector<int> G[MAX_N];
int nowi;

void dfs(int u,int fa,int s){
	nowi++;
	if(s==n){
		for(auto v:G[u]){
			if(v==fa)
				continue;
			mp[{min(u,v),max(u,v)}]=nowi+n;
			a[v]=nowi;
			dfs(v,u,0);
		}
	}
	if(s==0){
		for(auto v:G[u]){
			if(v==fa)
				continue;
			mp[{min(u,v),max(u,v)}]=nowi;
			a[v]=nowi+n;
			dfs(v,u,n);
		}
	}
} 

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		cin>>n;
		n=1<<n;
		nowi=0;
		for(int i=1;i<=n;i++)
			G[i].clear();
		edge.clear();
		mp.clear();
		for(int i=1;i<n;i++){
			int u,v;
			cin>>u>>v;
			G[u].push_back(v);
			G[v].push_back(u);
			edge.push_back({min(u,v),max(u,v)});
		}
		a[1]=n;
		dfs(1,0,n);
		cout<<1<<'\n';
		for(int i=1;i<=n;i++)
			cout<<a[i]<<' ';
		cout<<'\n';
		for(auto i:edge)
			cout<<mp[i]<<' ';
		cout<<'\n';
	}
	return 0;
}
