Codeforces Round #801 (Div. 2) D1D2. Tree Queries 题解

D1. Tree Queries (Easy Version)

D2. Tree Queries (Hard Version)

题意

给定一棵nnn个节点的无根树，你需要选择kkk个节点，使得在得知任意节点到这kkk个节点的路径边数后，可以确定该节点的编号，求最小的kkk值。

输入：t,n,x,y(1≤t≤104,1≤n≤(2000−easy/2×105−hard),1≤x,y≤n)t,n,x,y(1\leq t\leq 10^4,1\leq n\leq (2000-easy/2\times10^5-hard),1\leq x,y\leq n)t,n,x,y(1≤t≤104,1≤n≤(2000−easy/2×105−hard),1≤x,y≤n)

输出：最小需要选择多少个节点能确定所有节点

思路

对于任意节点，其子树最多只能有一棵完全不选择节点。

dfsdfsdfs遍历从根出发，遍历每个节点时，加上其所有子树答案，同时统计其子树答案为000的棵树(完全不选择节点的子树)，再加上棵数−1-1−1(只有一棵能完全不选择节点)，即

ans[x]=∑ans[i]+max(0,cnt−1),i∈G[x]ans[x]=\sum ans[i]+max(0,cnt-1),i\in G[x]ans[x]=∑ans[i]+max(0,cnt−1),i∈G[x]

简单版：枚举每个节点作为根的情况进行树上dpdpdp，最后加上111，即根自己。

困难版：枚举任意一个度≥3\geq3≥3的节点作为根的情况进行树上dpdpdp，因为该节点有两棵子树中有选择的节点，所以根不需要被选择；如果没有度≥3\geq3≥3的节点，只需要111个选择点即可(二叉树)。

复杂度：O(n2)−easy,O(n)−hardO(n^2)-easy,O(n)-hardO(n2)−easy,O(n)−hard

代码

D1 easy ver.
#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = 2010;

int T;
int n;
vector<int> G[MAX_N]; 

int dfs(int u,int fa){
	int s=0,cnt=0;
	for(auto i:G[u]){
		if(i==fa)
			continue;
		int res=dfs(i,u);
		s+=res;
		//统计子树个数 
		if(res==0)
			cnt++;
	}
	//返回所有子树和+子树个数-1（因为只能有一个子树不放结点 
	return s+max(0,cnt-1);
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		cin>>n;
		for(int i=1;i<=n;i++)
			G[i].clear();
		for(int i=1;i<n;i++){
			int u,v;
			cin>>u>>v;
			G[u].push_back(v);
			G[v].push_back(u);
		}
		if(n==1){
			cout<<0<<'\n';
			continue;
		}
		int ans=n;
		for(int i=1;i<=n;i++)
			ans=min(ans,dfs(i,i)+1);
		cout<<ans<<'\n';
	}
	return 0;
}

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = 200010;

int T;
int n;
vector<int> G[MAX_N]; 

int dfs(int u,int fa){
	int s=0,cnt=0;
	for(auto i:G[u]){
		if(i==fa)
			continue;
		int res=dfs(i,u);
		s+=res;
		//统计子树个数 
		if(res==0)
			cnt++;
	}
	//返回所有子树和+子树个数-1（因为只能有一个子树不放结点 
	return s+max(0,cnt-1);
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		cin>>n;
		for(int i=1;i<=n;i++)
			G[i].clear();
		for(int i=1;i<n;i++){
			int u,v;
			cin>>u>>v;
			G[u].push_back(v);
			G[v].push_back(u);
		}
		int ma=0;
		for(int i=1;i<=n;i++){
			int si=G[i].size();
			ma=max(ma,si); 
		}
		if(n==1){
			cout<<0<<'\n';
			continue;
		}
		if(ma<3){
			cout<<1<<'\n';
			continue;
		}
		for(int i=1;i<=n;i++)
			if(G[i].size()>=3){
				cout<<dfs(i,i)<<'\n';
				break;
			}
	}
	return 0;
}
