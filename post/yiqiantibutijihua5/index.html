Codeforces Round #802 (Div. 2) E. Serega the Pirate 题解

E. Serega the Pirate

题意

给定一个n∗mn*mn∗m的表格，包含了[1,n∗m][1,n*m][1,n∗m]内的每个数。

你需要从111出发，每次可以遍历上下左右(不越界)四个格子，如果所遍历的格子的值小于当前值或比遍历过的最大值大111，则可以访问，若最终能遍历到n∗mn*mn∗m所在格子，则遍历序列合法。

现给你一种操作，可以将任意两个格子的数字交换，使得最终表格存在一种合法遍历序列，问需要交换多少次。

输入：n,m,ai,j(1≤n,m≤400000,1≤ai,j≤n∗m)n,m,a_{i,j}(1\leq n,m\leq 400000,1\leq a_{i,j}\leq n*m)n,m,ai,j​(1≤n,m≤400000,1≤ai,j​≤n∗m)

输出：如果不需要交换，输出000；如果只需要交换一次，输入有多少种交换次数使表格合法；如果交换次数≥2\geq2≥2，输出222

思路

check(x,y)check(x,y)check(x,y)函数，判断ax,ya_{x,y}ax,y​是否能向相邻格子位移：只需要判断相邻格子中有没有比ax,ya_{x,y}ax,y​值小的即可。

首先对所有格子进行checkcheckcheck，同时记录坏点(无法位移至相邻格子的点)，如果坏点个数为000，则不需要交换；如果坏点个数>5>5>5，则至少需要两次，因为一次交换最多只能使555个坏点变好；如果坏点个数介于[1,4][1,4][1,4]之间，我们判断是否能通过一次交换使得表格合法。

首先将第一个坏点及其相邻点存入一个数组cndcndcnd，并将这些点标记为已访问，容易想到，要想一次交换使表格合法，有以下两种情况：

  1. 交换第一个坏点和表格其它点，再判断被影响的点和原本的坏点的好坏，如果全变好，说明交换使表格合法
  2. 交换第一个坏点的相邻点和表格其它点，再判断被影响的点和原本的坏点的好坏，如果全变好，说明交换使表格合法

被影响的点，也即交换的两个点的所有相邻点。

需要注意，两种情况有交集，因为坏点会和所有点交换，所以坏点的邻点在和所有点交换时需要跳过坏点。

复杂度：O(nm)O(nm)O(nm)

cndcndcnd数组最多555个点，将cndcndcnd数组的所有点和整个表格交换的复杂度是O(nm)O(nm)O(nm)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = 400010;

int n,m;
vector<vector<int> > a;
vector<int> bad;
vector<int> cnd;
int vis[MAX_N];

int dx[4]={1,-1,0,0},dy[4]={0,0,1,-1}; 

int check(int x,int y){
	if(a[x][y]==1)
		return 1;
	for(int i=0;i<4;i++){
		int nx=x+dx[i],ny=y+dy[i];
		if(nx<0||nx>=n||ny<0||ny>=m)
			continue;
			if(a[x][y]>a[nx][ny])
				return 1;
	}
	return 0;
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>n>>m;
	a.resize(n);
	for(auto&i:a)
		i.resize(m);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>a[i][j];
	//记录所有坏点（四周都比该点大） 
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(check(i,j)==0)
				bad.push_back(i*m+j);
	//如果没有坏点，答案为0 
	if(bad.size()==0){
		cout<<0<<'\n';
		return 0;
	}
	//如果坏点超过5个，答案至少为2 
	if(bad.size()>5){
		cout<<2<<'\n';
		return 0;
	}
	//把第一个坏点及其邻点全存进cnd数组，并标记为已访问 
	int now=bad[0];
	cnd.push_back(now);
	vis[now]=1;
	for(int i=0;i<4;i++){
		int nx=now/m+dx[i],ny=now%m+dy[i];
		if(nx<0||nx>=n||ny<0||ny>=m)
			continue;
		cnd.push_back(nx*m+ny);
		vis[nx*m+ny]=1;
	}
	//统计合法交换对数 
	int cnt=0;
	//遍历cnd的所有点 
	for(auto pos:cnd){
		//遍历所有节点 
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++){
				//如果该点已访问并且值不比pos的值大，跳过 
				if(vis[i*m+j]&&pos>=i*m+j)
					continue;
				//否则交换该点和pos 
				swap(a[pos/m][pos%m],a[i][j]);
				//判断交换后是否均不为坏点 
				int f=check(i,j)&&check(pos/m,pos%m);
				//遍历所有坏点，判断是否仍是坏点 
				for(auto t:bad)
					f&=check(t/m,t%m);
				//遍历pos和(i,j)交换后周围点是否为坏点 
				for(int k=0;k<4;k++){
					int x=pos/m+dx[k],y=pos%m+dy[k];
					if(f&&x>=0&&x<n&&y>=0&&y<m&&check(x,y)==0)
						f=0;
					x=i+dx[k],y=j+dy[k];
					if(f&&x>=0&&x<n&&y>=0&&y<m&&check(x,y)==0)
						f=0;
				}
				//回溯 
				swap(a[pos/m][pos%m],a[i][j]);
				//统计
				cnt+=f; 
			}
	}
	//如果存在合法交换，则输出合法交换次数 
	if(cnt)
		cout<<"1 "<<cnt<<'\n';
	//否则输出2 
	else
		cout<<2<<'\n';
	return 0;
}
