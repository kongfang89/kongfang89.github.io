E. AND-MEX Walk

题意

给定一个带权无向连通图，有nnn个顶点和mmm条边。从顶点uuu到vvv的路径被定义为从uuu开始以vvv结束的顶点序列p1,p2,…,pkp_1,p_2,\dots,p_kp1​,p2​,…,pk​，其中pi,pj(i≠j)p_i,p_j(i\neq j)pi​,pj​(i​=j)不一定不同。

路径长度定义如下：对于某一路径序列对应的权值数组w1,w2,…,wk−1w_1,w_2,\dots,w_{k-1}w1​,w2​,…,wk−1​，写下其前缀按位与数组，路径长度为该数组的MEXMEXMEX，即MEX({w1,w1&w2,…,w1&w2&…&wk−1})MEX(\{w_1,w_1\&w_2,\dots,w_1\&w_2\&\dots\&w_{k-1}\})MEX({w1​,w1​&w2​,…,w1​&w2​&…&wk−1​})。

现给定qqq个查询，每次给出两个顶点编号u,vu,vu,v，找到从uuu到vvv的最小路径长度。

输入：n,m,a,b,w,q,u,v(2≤n≤105,n−1≤m≤min(n(n−1)2,105),1≤a,b≤n,0≤w≤230,1≤q≤105,1≤u,v≤n)n,m,a,b,w,q,u,v(2\leq n\leq 10^5,n-1\leq m\leq min(\frac{n(n-1)}{2},10^5),1\leq a,b\leq n,0\leq w\leq 2^{30},1\leq q\leq10^5,1\leq u,v\leq n)n,m,a,b,w,q,u,v(2≤n≤105,n−1≤m≤min(2n(n−1)​,105),1≤a,b≤n,0≤w≤230,1≤q≤105,1≤u,v≤n)

输出：对于每个查询，输出其答案

思路

结论：MEXMEXMEX只会是0,1 or 20,1\space or\space20,1 or 2

证明：首先按位与数组一定是单调不增的，如果当前值为333，则与上任何值都只会变成3/2/1/03/2/1/03/2/1/0；如果当前值为222，则与上任何值都只会变成2/02/02/0；如果当前值为111，则与上任何值都只会变成1/01/01/0。

故MEXMEXMEX只会是0,1 or 20,1\space or\space20,1 or 2。

**Case 1 : MEX=0MEX=0MEX=0 **

若使MEX=0MEX=0MEX=0，则对于从uuu到vvv的路径，所有权值必须在同一位均不为000，则其前缀按位与数组每个元素均>0>0>0。

**Case 2 : MEX=1MEX=1MEX=1 **

若在路径中，对于任意非最低位，若uuu的连通块中有一条偶数边，则该连通块的按位与值会是大于000的偶数，这时我们再去找vvv，最终结果只会是>1/=0>1/=0>1/=0，则排除Case 1时MEX=1MEX=1MEX=1

**Case 3 : MEX=2MEX=2MEX=2 **

除了上述两种情况，MEX=2MEX=2MEX=2

复杂度：O((nmlogw+q)logw)O((nmlogw+q)logw)O((nmlogw+q)logw)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = 3e5+10;

struct union_find
{
    union_find(int n) : par_(n, -1) {}
    void init(int n)
    {
        par_.assign(n, -1);
    }
    int root(int x)
    {
        return par_[x] < 0 ? x : par_[x] = root(par_[x]);
    }
    bool unite(int x, int y) 
    {
        x = root(x);
        y = root(y);
        if (x == y) return false;
        if (par_[x] < par_[y])
        {
            par_[x] += par_[y];
            par_[y] = x;
        }
        else
        {
            par_[y] += par_[x];
            par_[x] = y;
        }
        return true;
    }
    bool same(int x, int y)
    {
        return root(x) == root(y);
    }
    int size(int x)
    {
        return -par_[root(x)];
    }
    vector<int> par_;
};

vector<pair<int,int> > G[MAX_N];
int u[MAX_N],v[MAX_N];
int good[MAX_N];

int main(){
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		int u,v,w;
		cin>>u>>v>>w;
		u--;
		v--;
		G[u].push_back({v,w});
		G[v].push_back({u,w});
		if(w%2==0){
			good[u]=1;
			good[v]=1;
		}
	}
	int q;
	cin>>q;
	for(int i=0;i<q;i++){
		cin>>u[i]>>v[i];
		u[i]--;
		v[i]--;
	}
	//默认答案全是2 
	vector<int> ans(q+10,2);
	//遍历每一位 
	for(int bit=0;bit<30;bit++){
		//构造并查集，判断有多少边权在当前位是1，将其加到同一集合 
		union_find uf(n+1);
		for(int u=0;u<n;u++)
			for(auto i:G[u]){
				int v=i.first,w=i.second;
				if(w&(1<<bit))
					uf.unite(u,v);
			}
		//遍历所有查询，如果起点和终点在同一集合，答案为0 
		for(int i=0;i<q;i++)
			if(uf.same(u[i],v[i]))
				ans[i]=min(ans[i],0);
		//如果是非零位，且存在偶数边，将偶数边和n+1加入同一集合 
		for(int i=0;i<n;i++)
			if(bit&&good[i])
				uf.unite(i,n);
		//如果u对应的集合有偶数边，则答案为1 
		for(int i=0;i<q;i++)
			if(uf.same(u[i],n))
				ans[i]=min(ans[i],1);
	}
	for(int i=0;i<q;i++)
		cout<<ans[i]<<'\n';
	return 0;
}
