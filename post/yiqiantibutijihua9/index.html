F. Tree Recovery

题意

给定一棵nnn个节点的树，并告知任意三点x,y,zx,y,zx,y,z的dis(x,z)dis(x,z)dis(x,z)与dis(y,z)dis(y,z)dis(y,z)是否相等，问是否能构造一棵树满足所有条件，如果可以则输出树的所有边。

输入：t,n,query[i][j][k](1≤t≤200,1≤n≤100,query∈(0,1))t,n,query[i][j][k](1\leq t\leq 200,1\leq n\leq 100,query\in(0,1))t,n,query[i][j][k](1≤t≤200,1≤n≤100,query∈(0,1))

输出：如果存在一棵树满足所有条件，输出Yes和这棵树的边；如果不存在输出No

思路

固定节点111作为树根，找到根节点的一个儿子，则所有边都可以求出来，再判断这棵树是否满足所有查询即可。

复杂度：O(n4)O(n^4)O(n4)

代码

#include <bits/stdc++.h>
#define INF INT_MAX
#define LINF LLONG_MAX 
#define endl '\n' 
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int MAX_N = 110;

int T;
int n;
//query[k][i][j]为1表示d(i,k)=d(j,k) 
bool query[MAX_N][MAX_N][MAX_N];
//深度
int d[MAX_N];
//dfs计算树G中所有节点深度 
void dfs(int u,int fa,int dep,vector<vector<int> > G){
	d[u]=dep;
	for(auto v:G[u]){
		if(v==fa)
			continue;
		dfs(v,u,dep+1,G);
	}
}

bool check(vector<vector<int> > G){
	int n=G.size();
	bool f=true;
	//遍历每个节点 
	for(int k=0;k<n&&f;k++){
		memset(d,-1,sizeof(d));
		//计算以该节点为根的深度数组 
		dfs(k,-1,0,G);
		//遍历其他节点 
		for(int i=0;i<n&&f;i++){
			if(d[i]==-1)
				f=false;
			//检查所有节点到根k的距离（深度）是否相等 
			for(int j=i+1;j<n&&f;j++)
				if((d[i]==d[j])!=(query[k][i][j]))
					f=false;
		}
	}
	return f;
}

int main(){
	freopen("1.in", "r", stdin);
	ios_base::sync_with_stdio(false);
	cout.tie(0);
	cin.tie(0);
	//freopen("1.out", "w", stdout);
	cin>>T;
	while(T--){
		cin>>n;
		for(int i=0;i<n;i++)
			for(int j=i+1;j<n;j++)
				for(int k=0;k<n;k++){
					char c;
					cin>>c;
					if(c=='1')
						query[k][i][j]=query[k][j][i]=1;
					else
						query[k][i][j]=query[k][j][i]=0;
				}
		int flag=0;
		//以0为根，遍历所有节点作为其儿子的树 
		for(int i=1;i<n;i++){
			//树 
			vector<vector<int> > G(n);
			G[0].push_back(i);
			G[i].push_back(0);
			//父亲节点数组 
			vector<int> p(n,-1);
			p[0]=i;
			p[i]=0;
			//bfs
			queue<int> q;
			q.push(0);
			q.push(i);
			while(!q.empty()){
				//取出一个节点 
				int u=q.front();
				q.pop();
				//遍历所有节点 
				for(int v=1;v<n;v++){
					//如果遍历到已经遍历过的节点或dis(u,v)!=dis(u,p[u]) 
					if(p[v]!=-1||!query[u][v][p[u]])
						//跳过 
						continue;
					q.push(v);
					//记录其父节点，表示已经遍历过
					p[v]=u;
					//连边 
					G[u].push_back(v);
					G[v].push_back(u);
				}
			}
			//检查树是否满足所有条件 
			if(check(G)){
				flag=1;
				cout<<"Yes\n";
				for(int i=0;i<n;i++){
					for(auto j:G[i]){
						if(i>=j)
							continue;
						cout<<i+1<<' '<<j+1<<'\n';
					}
				} 
				break;
			}
		}
		if(!flag)
			cout<<"No\n";
	}
	return 0;
}
